Решение системы линейных уравнений¶

$$ Ax = b$$
$$ \begin{bmatrix} 1 & 1 & 0 \\ 0 & 1 & 0 \\ 0 & 1 & 1 \\ \end{bmatrix} \ x = \ \begin{bmatrix} 2 \\ 5 \\ 9 \\ \end{bmatrix}$$
In [50]:
np.linalg.solve(A, b)
Out[50]:
array([-3.,  5.,  4.])
In [51]:
np.matmul(A, np.linalg.solve(A, b))
Out[51]:
array([2., 5., 9.])

x = np.linspace(-5, 5, 100)
y = x ** 2.

plt.title("$y = x^2$")

plt.ylabel("x")
plt.show();
